jQuery 文档处理

1.内部插入

append( )---appendTo( )

        $("#box").append("</p>一处水水水水</p>"); //都是在#box里面的最后添加</p>
        $("</p>一处水水水水</p>").appendTo("#box");

prepend( )---prependTo( )

        $("#box").prepend("</p>一处水水水水</p>"); //都是在#box里面的最前添加</p>
        $("</p>一处水水水水</p>").prependTo("#box");
2.外部插入

after( )---insertAfter( )

        $("#box").after("</p>一处水水水水</p>"); //都是在#box的后面紧跟着添加</p>
        $("</p>一处水水水水</p>").insertAfter("#box");

before( )---insertBefore( )

        $("#box").before("</p>一处水水水水</p>");  //都是在#box的前面紧跟着添加</p>
        $("</p>一处水水水水</p>").insertBefore("#box");
3.包裹

wrap( )---unwrap( )

       $("#box").wrap("</p>一处水水水水</p>");  // 给#box添加父级
       $("#ps").unwrap();  // 只删除#ps的父级，但父级里的元素依然存在
4.删除

empty( )

       $("#box").empty();  //把#box的内容清空

remove( )

       $("#box").remove();  //把整个#box移除

detach( )：与remove( )不同的是，所有绑定的事件、附加的数据等都会保留下来。

       $("p").detach(".hello");  //把所有p标签删除
       $("p").detach();          //把带有hello类的p标签删除
5.克隆

clone( )

       $("#box").append($("#box").clone());
       //clone要和.append等合用，选择要clone的对象，然后把它显示到某个位置
