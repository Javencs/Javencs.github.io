    <script>
        //播放、暂停切换
        var play = document.getElementById('play');
        var video = document.getElementById('video');
        play.onclick = function clicks(){
            if(video.paused){
                video.play();
                con.innerHTML = '暂停';
            }else{
                video.pause();
                con.innerHTML  = '播放';
            }
        }
    </script>
